RUN!
ID:PECL PACKAGES
DESCRIPTION:Installs all common PECL (PHP) packages
PROVIDES:php
BOUND:---
---
-pecl install memcache
@[ -f /usr/lib/php/modules/memcache.so ] || (echo "php memcache not installed"; exit 1)
@echo "extension=memcache.so" > /etc/php.d/memcache.ini

-pecl install ssh2-0.11.0
@[ -f /usr/lib/php/modules/ssh2.so ] || (echo "php ssh2 not installed"; exit 1)
@echo "extension=ssh2.so" > /etc/php.d/ssh2.ini
---
